import csv
import pandas as pd

# Read the CSV file
data = pd.read_csv("C:\Users\SaiRamaKrishnaNampal\Downloads\fictional_richest_people.xlsx")

# Clean and preprocess the data
data['Net Worth'] = data['Net Worth'].str.replace('$', '').str.replace(' billion', '').astype(float)

# Find the richest person
richest_index = data['Net Worth'].idxmax()
richest_person = data.loc[richest_index, 'Name']
richest_net_worth = data.loc[richest_index, 'Net Worth']

# Print the result
print(f"The richest person is {richest_person} with a net worth of ${richest_net_worth} billion.")
